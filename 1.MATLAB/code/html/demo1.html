VI. MATLAB基本绘图操作

0. 绘图

x = (0:0.01:0.98)
y = sin(2*pi*x*8)
plot(x,y)

hold on;            % 在原图像上继续画

y2 = cos(2*pi*x*8)
plot(x,y2,'r')

axis([30, 100, 30, 100])	% 设置横纵坐标的量度

xlabel('time')          % x轴名称
ylabel('value')         % y轴名称
legend('sin','cos')     % 每条曲线的含义
title('my plot')
print -dpng 'myPlot.png'

close                   % 关闭窗口

1. 二维平面绘图

x = 0:0.01:2*pi;
y = sin(x);
figure
plot(x, y)
title('y = sin(x)')
xlabel('x')
ylabel('sin(x)')
xlim([0 2*pi])

x = 0:0.01:20;
y1 = 200*exp(-0.05*x).*sin(x);
y2 = 0.8*exp(-0.5*x).*sin(10*x);
figure
[AX,H1,H2] = plotyy(x,y1,x,y2,'plot');
set(get(AX(1),'Ylabel'),'String','Slow Decay')
set(get(AX(2),'Ylabel'),'String','Fast Decay')
xlabel('Time (\musec)')
title('Multiple Decay Rates')
set(H1,'LineStyle','--')
set(H2,'LineStyle',':')

2. 三维立体绘图

t = 0:pi/50:10*pi;
plot3(sin(t),cos(t),t)
xlabel('sin(t)')
ylabel('cos(t)')
zlabel('t')
grid on
axis square

VII. MATLAB文件导入

1. mat格式

save data.mat x y1 y2
clear all
load data.mat

2. txt格式

M = importdata('myfile.txt');

S = M.data;
save 'data.txt' S -ascii
T = load('data.txt');

isequal(S, T)

3. xls格式

xlswrite('data.xls',S)
W = xlsread('data.xls');
isequal(S, W)

xlswrite('data.xlsx',S)
U = xlsread('data.xlsx');
isequal(S, U)

4. csv格式

csvwrite('data.csv',S)
V = csvread('data.csv');
isequal(S, V)
